pragma solidity 0.8.24;

/// @title EntitlementRegistry â€” paid-balance ledger for Pangolin's funder service
/// @notice MVP-2 issue 2.2. Records per-user paid credit balance + per-user nonce.
/// @dev Cardinal contract design rules (locked at 2.1's L2; inherited verbatim):
///   - No admin, no owner, no upgrade hooks, no `pause`, no `selfdestruct`,
///     no `delegatecall`, no `receive` / `fallback`, non-`payable`.
///   - Signer authorities (paymentAuthority, redemptionAuthority â€” per Q-a)
///     are constructor-immutable. A compromised key is fixed by deploying a
///     v2 contract, NOT by an in-place setter.
contract EntitlementRegistry {
    // -------------------------------------------------------------------
    // EIP-712 / signing constants
    // -------------------------------------------------------------------

    bytes32 private constant EIP712_DOMAIN_TYPEHASH = keccak256(
        "EIP712Domain(string name,string version,uint256 chainId,address verifyingContract)"
    );

    /// @dev Credit struct typehash. Payment processor signs this.
    /// Six fields: userId (bytes32 â€” per Q-b decision), amount (uint256),
    /// nonce (uint64), schemaVersion (uint16), expiresAt (uint64 â€” anti-stale
    /// signature; the payment processor's signed credits expire after some
    /// window).
    bytes32 private constant CREDIT_TYPEHASH = keccak256(
        "Credit(bytes32 userId,uint256 amount,uint64 nonce,uint16 schemaVersion,uint64 expiresAt)"
    );

    /// @dev Redemption struct typehash. Funder authority signs this.
    /// Same shape â€” different domain field (per Q-a/Q-c).
    bytes32 private constant REDEMPTION_TYPEHASH = keccak256(
        "Redemption(bytes32 userId,uint256 amount,uint64 nonce,uint16 schemaVersion,uint64 expiresAt)"
    );

    uint16 public constant MAX_KNOWN_SCHEMA_VERSION = 1;

    // -------------------------------------------------------------------
    // Immutable storage (set at deploy time; cannot change)
    // -------------------------------------------------------------------

    /// @dev The single address authorized to sign Credit attestations.
    address public immutable PAYMENT_AUTHORITY;

    /// @dev The single address authorized to sign Redemption attestations.
    /// Per Q-a may equal PAYMENT_AUTHORITY (single-signer model) or differ
    /// (split-trust model). Resolved at deploy.
    address public immutable REDEMPTION_AUTHORITY;

    /// @dev EIP-712 v4 domain separator, computed once at construction.
    bytes32 public immutable DOMAIN_SEPARATOR;

    // -------------------------------------------------------------------
    // Mutable storage (only writes happen via the two typed entry points)
    // -------------------------------------------------------------------

    /// @dev Per-user paid credit balance. Unit is abstract "credits";
    ///      the funder service maps credits â†’ revisions per the pricing
    ///      spec (5,000 credits per $30 today; future repricing doesn't
    ///      touch the contract). uint256.
    mapping(bytes32 => uint256) public balance;

    /// @dev Per-user monotonic nonce. Increments on every successful
    ///      credit OR redemption. Replay-protection primitive.
    mapping(bytes32 => uint64) public nonce;

    // -------------------------------------------------------------------
    // Custom errors (cheaper than string reverts; audit-friendly)
    // -------------------------------------------------------------------

    error ErrInvalidSignature();
    error ErrUnauthorizedSigner();
    error ErrInsufficientBalance();
    error ErrNonceTooLow();
    error ErrUnsupportedSchemaVersion();
    error ErrAttestationExpired();
    error ErrZeroAuthority();

    // -------------------------------------------------------------------
    // Events
    // -------------------------------------------------------------------

    event Credited(
        bytes32 indexed userId,
        uint256 amount,
        uint256 newBalance,
        uint64 nonce,
        uint16 schemaVersion
    );
    event Redeemed(
        bytes32 indexed userId,
        uint256 amount,
        uint256 newBalance,
        uint64 nonce,
        uint16 schemaVersion
    );

    // -------------------------------------------------------------------
    // Constructor
    // -------------------------------------------------------------------

    constructor(address paymentAuthority_, address redemptionAuthority_) {
        if (paymentAuthority_ == address(0)) revert ErrZeroAuthority();
        if (redemptionAuthority_ == address(0)) revert ErrZeroAuthority();
        PAYMENT_AUTHORITY = paymentAuthority_;
        REDEMPTION_AUTHORITY = redemptionAuthority_;
        DOMAIN_SEPARATOR = keccak256(
            abi.encode(
                EIP712_DOMAIN_TYPEHASH,
                keccak256(bytes("Pangolin EntitlementRegistry")),
                keccak256(bytes("1")),
                block.chainid,
                address(this)
            )
        );
    }

    // -------------------------------------------------------------------
    // External entry points
    // -------------------------------------------------------------------

    function credit(
        bytes32 userId,
        uint256 amount,
        uint64 attestationNonce,
        uint16 schemaVersion,
        uint64 expiresAt,
        bytes calldata signature
    ) external returns (uint256 newBalance) {
        // 1. Bound the schema version BEFORE any work.
        if (schemaVersion > MAX_KNOWN_SCHEMA_VERSION) revert ErrUnsupportedSchemaVersion();

        // 2. Expiry check (anti-stale-signature; protects against a long-stored
        //    leaked signature being submitted years later).
        if (block.timestamp > expiresAt) revert ErrAttestationExpired();

        // 3. EIP-712 digest + signature recover.
        bytes32 digest = _hashCredit(userId, amount, attestationNonce, schemaVersion, expiresAt);
        address signer = _recover(digest, signature);
        if (signer == address(0)) revert ErrInvalidSignature();
        if (signer != PAYMENT_AUTHORITY) revert ErrUnauthorizedSigner();

        // 4. Nonce check â€” strict equality (per Q-c-2 if "exact" is locked) or
        //    `>=` (if "monotonic-allow-skip" is locked).
        if (attestationNonce != nonce[userId]) revert ErrNonceTooLow();

        // 5. All checks passed; apply state change.
        unchecked {
            nonce[userId] = attestationNonce + 1;
            newBalance = balance[userId] + amount;
        }
        balance[userId] = newBalance;
        emit Credited(userId, amount, newBalance, attestationNonce, schemaVersion);
    }

    function redeem(
        bytes32 userId,
        uint256 amount,
        uint64 attestationNonce,
        uint16 schemaVersion,
        uint64 expiresAt,
        bytes calldata signature
    ) external returns (uint256 newBalance) {
        // 1. Schema version bound.
        if (schemaVersion > MAX_KNOWN_SCHEMA_VERSION) revert ErrUnsupportedSchemaVersion();

        // 2. Expiry check.
        if (block.timestamp > expiresAt) revert ErrAttestationExpired();

        // 3. EIP-712 digest + signature recover.
        bytes32 digest = _hashRedemption(userId, amount, attestationNonce, schemaVersion, expiresAt);
        address signer = _recover(digest, signature);
        if (signer == address(0)) revert ErrInvalidSignature();
        if (signer != REDEMPTION_AUTHORITY) revert ErrUnauthorizedSigner();

        // 4. Nonce check.
        if (attestationNonce != nonce[userId]) revert ErrNonceTooLow();

        // 5. Balance check (no underflow on `unchecked` would burn credits).
        if (balance[userId] < amount) revert ErrInsufficientBalance();

        // 6. Apply state change.
        unchecked {
            nonce[userId] = attestationNonce + 1;
            newBalance = balance[userId] - amount;
        }
        balance[userId] = newBalance;
        emit Redeemed(userId, amount, newBalance, attestationNonce, schemaVersion);
    }

    // -------------------------------------------------------------------
    // View helpers (off-chain digest oracle for the signing services)
    // -------------------------------------------------------------------

    function hashCredit(
        bytes32 userId,
        uint256 amount,
        uint64 attestationNonce,
        uint16 schemaVersion,
        uint64 expiresAt
    ) external view returns (bytes32) {
        return _hashCredit(userId, amount, attestationNonce, schemaVersion, expiresAt);
    }

    function hashRedemption(
        bytes32 userId,
        uint256 amount,
        uint64 attestationNonce,
        uint16 schemaVersion,
        uint64 expiresAt
    ) external view returns (bytes32) {
        return _hashRedemption(userId, amount, attestationNonce, schemaVersion, expiresAt);
    }

    // -------------------------------------------------------------------
    // Internal helpers
    // -------------------------------------------------------------------

    function _hashCredit(
        bytes32 userId,
        uint256 amount,
        uint64 attestationNonce,
        uint16 schemaVersion,
        uint64 expiresAt
    ) internal view returns (bytes32) {
        bytes32 structHash = keccak256(
            abi.encode(
                CREDIT_TYPEHASH,
                userId,
                amount,
                attestationNonce,
                schemaVersion,
                expiresAt
            )
        );
        return keccak256(abi.encodePacked("\x19\x01", DOMAIN_SEPARATOR, structHash));
    }

    function _hashRedemption(
        bytes32 userId,
        uint256 amount,
        uint64 attestationNonce,
        uint16 schemaVersion,
        uint64 expiresAt
    ) internal view returns (bytes32) {
        bytes32 structHash = keccak256(
            abi.encode(
                REDEMPTION_TYPEHASH,
                userId,
                amount,
                attestationNonce,
                schemaVersion,
                expiresAt
            )
        );
        return keccak256(abi.encodePacked("\x19\x01", DOMAIN_SEPARATOR, structHash));
    }

    function _recover(bytes32 digest, bytes calldata signature) internal pure returns (address) {

        // (Body identical to RevisionLogV1's `_recover` â€” copy verbatim.)

### 2. Signature verification + replay-protection flow

For both `credit` and `redeem`:

1. Bound `schemaVersion` against `MAX_KNOWN_SCHEMA_VERSION = 1`.
2. Check `block.timestamp <= expiresAt` (anti-stale-signature).
3. Compute the EIP-712 digest via the appropriate typehash + `DOMAIN_SEPARATOR`.
4. Recover the signer via the same `_recover` helper as `RevisionLogV1.sol` (verbatim â€” see L4); guards against ecrecover malleability + `address(0)` sentinel.
5. Compare the recovered signer against the appropriate immutable authority (`PAYMENT_AUTHORITY` for credits, `REDEMPTION_AUTHORITY` for redemptions). Mismatch â†’ `ErrUnauthorizedSigner`.
6. Check `attestationNonce == nonce[userId]` (per Q-c-2 if "exact" is locked) â€” prevents replay AND out-of-order submission.
7. For `redeem`: check `balance[userId] >= amount` (else `ErrInsufficientBalance`).
8. Apply state change: `nonce[userId] = attestationNonce + 1`, then update balance, then emit.

**Replay across deployments / chains is defeated by `DOMAIN_SEPARATOR`** binding `chainId` + `address(this)` (same as v1's R-a Implementation impact (4)).

**Replay across credit/redemption is defeated by the typehash** â€” a signed `Credit` cannot be submitted as a `Redemption` because the typehash bytes differ â†’ the digest differs â†’ `ecrecover` recovers a different (random) address â†’ not equal to the expected authority â†’ `ErrUnauthorizedSigner`.

**Replay across users is defeated by `userId` being in the struct hash.**

**Replay across nonces is defeated by the strict-equality nonce check** â€” per Q-c-2.

### 3. Threat-model interactions

- **Invariant #1 (no plaintext on chain)** â€” preserved. The registry stores a balance integer + a nonce â€” neither is "plaintext credential data". The `userId` per Q-b is either an Ethereum address (already public per D-006's mitigation) or an opaque `bytes32` (no information without the off-chain mapping). NO vault data, no credential data, no key material on chain.
- **Invariant #2 (relay/funder/indexer cannot forge revisions)** â€” strengthened. The registry's *credit* path enforces "only the payment-authority key can inflate balances"; the *redeem* path enforces "only the redemption-authority key can deflate balances". A compromised funder cannot credit itself with new balances (only deflate); a compromised payment processor cannot deflate (only inflate). **This is the split-trust security model from Q-a Option B**.
- **Invariant #5 (session authority never on-chain)** â€” preserved. The entitlement-authority is NOT a session authority; it's a separate concept (payment processor vs user session). The wording in `THREAT_MODEL.md` #5 stays clean.
- **Invariant #11 (Â§18.7 ladder)** â€” preserved + extended. The contract's `schemaVersion` bound + the event's unindexed `schemaVersion` field both follow the policy. Future v1.1 attestations land on `ErrUnsupportedSchemaVersion`.
- **Invariant #12 (AAD coverage)** â€” preserved. The EIP-712 typed-data digest binds every disambiguating byte (typehash, userId, amount, nonce, schemaVersion, expiresAt, chainId-via-domain, contract-address-via-domain). A flipped byte â†’ digest mismatch â†’ signer recovery mismatches the expected authority â†’ `ErrUnauthorizedSigner`. Same posture as `tampered_header_byte_fails_auth` in 1.10's export AEAD.

### 4. Privacy posture (D-005, D-006, Â§8.3, Â§11.1 risk row)

**What a chain observer learns about a user:** the `userId` and the credit + redemption events for that user. Specifically: total credited, total redeemed, current balance, transaction count (nonce).

The privacy implication depends on Q-b:
- If `userId == device wallet address` â†’ an observer who knows a device's address knows that user's full payment history. This is the D-006 known mitigation (Enhanced Privacy Mode addresses it in Phase 2).
- If `userId == opaque bytes32` â†’ an observer sees the events but cannot trivially correlate a `userId` to an address or a user identity. The payment processor + funder service know the mapping; chain observers don't.

**The funder service** (issue 3.4) learns the `userId` for every redemption it signs â€” by definition, it dispenses ETH to a device address, so it knows the device-address-to-userId mapping. This is the Â§17 funder operations note: *"address-vs-payment correlation policy"* â€” the funder logs minimal data, retains for limited window for abuse protection only, no account linkage. **Privacy-mode users (Phase 2) can route top-ups through mixer before use.**

**Payment processor** has the userId-to-payment-history mapping by definition (it issues the signed credits). This is also off-chain and out of MVP scope.

## Open questions for Kelvin

### Q-a Â· Signer authority model â€” one key or two?

The master plan Â§4 row 2.2 says "credits via signed payment proof; redemptions via signed top-up requests" â€” implies two distinct signed-message types, but doesn't pin whether they share a key.

Three plausible models:

**Option A: Single-signer (`PAYMENT_AUTHORITY` signs both Credit AND Redemption).** Simplest contract. One key to manage off-band. **Risk: a single key compromise drains every user's balance** AND inflates any user's balance â€” the attacker gets the full set of primitives.

**Option B: Split-signer (`PAYMENT_AUTHORITY` signs Credit; `REDEMPTION_AUTHORITY` signs Redemption).** What the contract sketch above defaults to. Splits trust: a compromised funder-service can ONLY decrement balances (deny-service against users; cannot inflate to drain the funder treasury). A compromised payment-processor can ONLY inflate balances (cannot decrement; cannot directly drain the funder treasury â€” but a credit inflation followed by the user's own redemption attestation is an indirect attack). **Recommended** â€” narrower trust surface, narrower compromise blast radius.

**Option C: User-signed redemption (`PAYMENT_AUTHORITY` signs Credit; the USER's device-wallet key signs Redemption).** The user pre-authorizes the funder to decrement N credits when the user wants gas. The funder submits the user-signed attestation. **Trade-off: more UX friction (user must approve each redemption â€” possibly via a periodic batch-authorization)**; but eliminates the entire funder-key-compromise category.

**Pick one.** Default unless told otherwise: Option B (split-signer).

### Q-b Â· `userId` type â€” Ethereum address or opaque bytes32?

The master plan says "per user account" â€” ambiguous. Three options:

**Option A: `address userId` (Ethereum address â€” same as the user's device-wallet secp256k1 address).** Simplest. **Privacy: chain observers can correlate device addresses with payment balances** â€” but D-006 already says device addresses are observable on-chain (mitigated by Phase-2 Enhanced Privacy Mode). So option A doesn't make privacy worse than the baseline.

**Option B: `bytes32 userId` (opaque Pangolin user identifier).** Decouples payment from device address. **Privacy advantage: chain observers see payment events but cannot trivially correlate `userId` to an address or a user identity** â€” the payment processor knows the mapping; chain observers don't. The funder service needs the mapping to dispatch ETH to the right device â€” but the funder service is Pangolin-operated, so it learns this anyway. So option B's privacy advantage is mostly against chain observers (which is real and useful).

**Option C: `bytes32 userId = keccak256(vault_id || user_salt)` (privacy-preserving but vault-tied).** A vault-scoped identifier. **Trade-off: complicates the multi-device model** â€” same user with two devices either has two `userId`s (one per vault) or shares one `userId` across vaults (in which case the userId is a per-account thing, not a per-device thing).

**Pick one.** Default unless told otherwise: Option B (`bytes32 userId`).

### Q-c Â· Redemption-flow exact wording â€” who signs the decrement?

The master plan says "redemptions decrement balance via **signed top-up requests**" â€” wording-ambiguous.

**Sub-question Q-c-1 (the direction):** is the signed message:
- (i) "I (funder) just dispensed X credits worth of gas to user U â€” please decrement their balance" (funder-push, Option B in Q-a), or
- (ii) "I (user) authorize the funder to dispense up to X credits when I want gas â€” decrement my balance now in exchange" (user-pull, Option C in Q-a), or
- (iii) "I (payment-processor) am decrementing user U's balance by X because their account was downgraded / refund / cancellation" (admin-style, NOT this MVP).

Resolves with Q-a.

**Sub-question Q-c-2 (the nonce-check shape):**
- Strict equality: `attestationNonce == nonce[userId]` â€” forces in-order submission; one signed attestation per slot.
- Loose (`>= nonce[userId]`): allows the signer to mint multiple attestations and the user/funder to submit them in any order; only the highest-nonce one ever lands (lower-nonce ones get rejected on submit).

Strict equality is simpler and matches v1's self-bootstrap "atomic ratchet" feel. Loose allows out-of-order submission which is operationally nice for the funder (it can mint a redemption attestation in advance and submit later). **Default unless told otherwise: strict equality.**

### Q-d Â· Confirm: signer keys are constructor-immutable, no rotation in v2.2

Per L8 + L2, the immutable `PAYMENT_AUTHORITY` + `REDEMPTION_AUTHORITY` cannot be changed in place. **If either key is compromised, the only fix is deploying a v2 contract with a new authority + telling everyone (off-band) to point at the new address.** This is the same posture as RevisionLogV1 / V0 â€” "no admin = redeploy is the only fix". Confirm.

### Q-e Â· Anti-stale-signature: include `expiresAt` in the signed payload?

The contract sketch above includes `uint64 expiresAt` in the EIP-712 struct + checks `block.timestamp <= expiresAt` at verify. This protects against:
- A signature leaked years ago being submitted today (if the signer key is later compromised, only attestations signed AFTER the compromise are exposed; pre-compromise attestations have already expired).
- A signature being held by an attacker who intercepts the user's connection and submits later when the user's nonce has advanced (but the nonce check already covers this; expiry is defence-in-depth).

**Trade-off: adds 8 bytes to the signed payload + a `block.timestamp` read.** Negligible cost. **Default: include `expiresAt`.**

The recommended expiry window (off-chain decision; not on-chain enforced beyond the field): **5 minutes** for redemption attestations (operational immediacy); **1 hour** for credit attestations (user might pay and submit minutes later via a separate device). The contract is agnostic to the window; the signer picks `expiresAt = now + window`.
